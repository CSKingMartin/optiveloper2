import Toggle from '@hanga/Editor/Toggle';

<StatefulContext.Provider>
  <StatefulContext.Consumer>
    {
      ({ isDisabled }) =>
        <React.Fragment>
          <Toggle defaultValue={false} name="isDisabled"></Toggle>
          <p>
            <button disabled={isDisabled}>
              This button is {isDisabled ? 'disabled' : 'enabled'}
            </button>
          </p>
        </React.Fragment>
    }
  </StatefulContext.Consumer>
</StatefulContext.Provider>
