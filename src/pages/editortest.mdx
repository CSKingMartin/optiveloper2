import Editor from '@hanga';

<StatefulContext.Provider>
  <StatefulContext.Consumer>
    {
      ({ isDisabled }) =>
        <React.Fragment>
          <Editor.Toggle defaultValue={false} name="isDisabled"></Editor.Toggle>
          <p>
            <button disabled={isDisabled}>
              This button is {isDisabled ? 'disabled' : 'enabled'}
            </button>
          </p>
        </React.Fragment>
    }
  </StatefulContext.Consumer>

  <StatefulContext.Consumer>
    {
      ({ activeText }) =>
        <React.Fragment>
          <Editor.Text defaultValue="Lorem hipsum" label="Active Text" name="activeText"></Editor.Text>
          <p>{activeText}</p>
        </React.Fragment>
    }
  </StatefulContext.Consumer>
  <StatefulContext.Consumer>
      {
        ({ multiColor }) =>
          <React.Fragment>
            <h2>Test</h2>
            <Editor.Select name="multiColor" label="Color" defaultValue={'green'} options={['red', 'blue', 'green']}></Editor.Select>
            <p style={{ color: multiColor }}>What</p>
          </React.Fragment>
      }
    </StatefulContext.Consumer>
</StatefulContext.Provider>
